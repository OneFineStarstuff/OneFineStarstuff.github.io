import numpy as np
def quarantine(g, history, z=3.5):
    norms = [np.linalg.norm(h) for h in history]
    mu, sd = np.mean(norms), np.std(norms) + 1e-9
    return abs((np.linalg.norm(g) - mu) / sd) >= z
