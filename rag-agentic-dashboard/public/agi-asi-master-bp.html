from oqs import Signature
import nacl.signing

def hybrid_sign(payload: bytes, ed_key, ml_key):
    ed_sig = ed_key.sign(payload).signature
    sig = Signature('ML-DSA-65')
    pq_sig = sig.sign(payload, ml_key)
    return ed_sig + b'||' + pq_sig
